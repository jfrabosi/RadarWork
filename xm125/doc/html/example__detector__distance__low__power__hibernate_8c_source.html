26 /** \example example_detector_distance_low_power_hibernate.c
27  * @brief This is an example on how to put the sensor in hibernation and the system in a low power state between measurements
28  * @n
29  * The example executes as follows:
30  * - Create a distance configuration
31  * - Create a sensor instance
32  * - Enable sensor
33  * - Create a detector instance
34  * - Calibrate the sensor
35  * - Calibrate the detector
36  * - Enter hibernation
37  * - Loop forever:
38  * - Exit hibernation
39  * - Prepare the detector
40  * - Perform a sensor measurement and read out the data
41  * - Enter hibernation
42  * - Process the measurement and get detector result
43  * - Put the system in deep sleep for a specified amount of time
44  * - Destroy the configuration
45  * - Destroy the detector instance
46  * - Destroy the sensor instance
47  */

78 int acconeer_main(int argc, char *argv[])
79 {
80  (void)argc;
81  (void)argv;
82  acc_detector_distance_config_t *distance_config = NULL;
83  acc_detector_distance_handle_t *distance_handle = NULL;
84  acc_sensor_t *sensor = NULL;
85  void *buffer = NULL;
86  uint8_t *detector_cal_result_static = NULL;
87  uint32_t buffer_size = 0U;
88  uint32_t detector_cal_result_static_size = 0U;
89  acc_detector_cal_result_dynamic_t detector_cal_result_dynamic = { 0 };
90 
91  printf("Acconeer software version %s\n", acc_version_get());
92 
93  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
94 
95  if (!acc_rss_hal_register(hal))
96  {
97  return EXIT_FAILURE;
98  }
99 
100  distance_config = acc_detector_distance_config_create();
101  if (distance_config == NULL)
102  {
103  printf("acc_detector_distance_config_create() failed\n");
104  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
105  return EXIT_FAILURE;
106  }
107 
108  set_config(distance_config, DISTANCE_PRESET_CONFIG_BALANCED);
109 
110  uint32_t sleep_time_ms = (uint32_t)(1000.0f / DEFAULT_UPDATE_RATE);
111 
112  acc_integration_set_periodic_wakeup(sleep_time_ms);
113 
114  distance_handle = acc_detector_distance_create(distance_config);
115  if (distance_handle == NULL)
116  {
117  printf("acc_detector_distance_create() failed\n");
118  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
119  return EXIT_FAILURE;
120  }
121 
122  if (!acc_detector_distance_get_sizes(distance_handle, &buffer_size, &detector_cal_result_static_size))
123  {
124  printf("acc_detector_distance_get_buffer_size() failed\n");
125  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
126  return EXIT_FAILURE;
127  }
128 
129  buffer = acc_integration_mem_alloc(buffer_size);
130  if (buffer == NULL)
131  {
132  printf("buffer allocation failed\n");
133  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
134  return EXIT_FAILURE;
135  }
136 
137  detector_cal_result_static = acc_integration_mem_alloc(detector_cal_result_static_size);
138  if (detector_cal_result_static == NULL)
139  {
140  printf("buffer allocation failed\n");
141  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
142  return EXIT_FAILURE;
143  }
144 
145  acc_hal_integration_sensor_supply_on(SENSOR_ID);
146  acc_hal_integration_sensor_enable(SENSOR_ID);
147 
148  sensor = acc_sensor_create(SENSOR_ID);
149  if (sensor == NULL)
150  {
151  printf("acc_sensor_create() failed\n");
152  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
153  return EXIT_FAILURE;
154  }
155 
156  // Calibrate sensor
157  bool status;
158  bool cal_complete = false;
159  acc_cal_result_t sensor_cal_result;
160 
161  do
162  {
163  status = acc_sensor_calibrate(sensor, &cal_complete, &sensor_cal_result, buffer, buffer_size);
164 
165  if (cal_complete)
166  {
167  break;
168  }
169 
170  if (status)
171  {
172  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
173  }
174  } while (status);
175 
176  if (!status)
177  {
178  printf("acc_sensor_calibrate() failed\n");
179  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
180  return EXIT_FAILURE;
181  }
182 
183  /* Reset sensor after calibration by disabling/enabling it */
184  acc_hal_integration_sensor_disable(SENSOR_ID);
185  acc_hal_integration_sensor_enable(SENSOR_ID);
186 
187  // Calibrate detector
188  bool done = false;
189 
190  do
191  {
192  status = acc_detector_distance_calibrate(sensor,
193  distance_handle,
194  &sensor_cal_result,
195  buffer,
196  buffer_size,
197  detector_cal_result_static,
198  detector_cal_result_static_size,
199  &detector_cal_result_dynamic,
200  &done);
201 
202  if (done)
203  {
204  break;
205  }
206 
207  if (status)
208  {
209  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
210  }
211  } while (status);
212 
213  if (!status)
214  {
215  printf("acc_detector_distance_calibrate() failed\n");
216  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
217  return EXIT_FAILURE;
218  }
219 
220  acc_hal_integration_sensor_disable(SENSOR_ID);
221 
222  acc_hal_integration_sensor_enable(SENSOR_ID);
223 
224  if (!acc_detector_distance_prepare(distance_handle, distance_config, sensor, &sensor_cal_result, buffer, buffer_size))
225  {
226  printf("acc_detector_presence_prepare() failed\n");
227  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
228  return EXIT_FAILURE;
229  }
230 
231  if (!acc_sensor_hibernate_on(sensor))
232  {
233  printf("acc_sensor_hibernate_on failed\n");
234  acc_sensor_status(sensor);
235  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
236  return EXIT_FAILURE;
237  }
238 
239  acc_hal_integration_sensor_disable(SENSOR_ID);
240 
241  while (true)
242  {
243  acc_detector_distance_result_t result;
244 
245  acc_hal_integration_sensor_enable(SENSOR_ID);
246 
247  if (!acc_sensor_hibernate_off(sensor))
248  {
249  printf("acc_sensor_hibernate_off failed\n");
250  acc_sensor_status(sensor);
251  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
252  return EXIT_FAILURE;
253  }
254 
255  bool result_available = false;
256 
257  do
258  {
259  if (!acc_detector_distance_prepare(distance_handle, distance_config, sensor, &sensor_cal_result, buffer,
260  buffer_size))
261  {
262  printf("acc_detector_distance_prepare() failed\n");
263  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
264  return EXIT_FAILURE;
265  }
266 
267  if (!acc_sensor_measure(sensor))
268  {
269  printf("acc_sensor_measure failed\n");
270  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
271  return EXIT_FAILURE;
272  }
273 
274  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
275  {
276  printf("Sensor interrupt timeout\n");
277  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
278  return EXIT_FAILURE;
279  }
280 
281  if (!acc_sensor_read(sensor, buffer, buffer_size))
282  {
283  printf("acc_sensor_read failed\n");
284  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
285  return EXIT_FAILURE;
286  }
287 
288  if (!acc_detector_distance_process(distance_handle, buffer, detector_cal_result_static, &detector_cal_result_dynamic,
289  &result_available,
290  &result))
291  {
292  printf("acc_detector_distance_process failed\n");
293  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
294  return EXIT_FAILURE;
295  }
296  } while (!result_available);
297 
298  if (!acc_sensor_hibernate_on(sensor))
299  {
300  printf("acc_sensor_hibernate_on failed\n");
301  acc_sensor_status(sensor);
302  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
303  return EXIT_FAILURE;
304  }
305 
306  acc_hal_integration_sensor_disable(SENSOR_ID);
307 
308  print_result(&result);
309 
310  acc_integration_sleep_until_periodic_wakeup();
311  }
312 
313  cleanup(distance_handle, distance_config, sensor, buffer, detector_cal_result_static);
314 
315  printf("Application finished OK\n");
316 
317  return EXIT_SUCCESS;
318 }

357 static void set_config(acc_detector_distance_config_t *detector_config, distance_preset_config_t preset)
358 {
359  // Add configuration of the detector here
360  switch (preset)
361  {
362  case DISTANCE_PRESET_CONFIG_NONE:
363  // Add configuration of the detector here
364  break;
365 
366  case DISTANCE_PRESET_CONFIG_BALANCED:
367  acc_detector_distance_config_start_set(detector_config, 0.25f);
368  acc_detector_distance_config_end_set(detector_config, 3.0f);
369  acc_detector_distance_config_max_step_length_set(detector_config, 0U);
370  acc_detector_distance_config_max_profile_set(detector_config, ACC_CONFIG_PROFILE_5);
371  acc_detector_distance_config_reflector_shape_set(detector_config, ACC_DETECTOR_DISTANCE_REFLECTOR_SHAPE_GENERIC);
372  acc_detector_distance_config_peak_sorting_set(detector_config, ACC_DETECTOR_DISTANCE_PEAK_SORTING_STRONGEST);
373  acc_detector_distance_config_threshold_method_set(detector_config, ACC_DETECTOR_DISTANCE_THRESHOLD_METHOD_CFAR);
374  acc_detector_distance_config_threshold_sensitivity_set(detector_config, 0.5f);
375  acc_detector_distance_config_signal_quality_set(detector_config, 15.0f);
376  acc_detector_distance_config_close_range_leakage_cancellation_set(detector_config, false);
377  break;
378 
379  case DISTANCE_PRESET_CONFIG_HIGH_ACCURACY:
380  acc_detector_distance_config_start_set(detector_config, 0.25f);
381  acc_detector_distance_config_end_set(detector_config, 3.0f);
382  acc_detector_distance_config_max_step_length_set(detector_config, 2U);
383  acc_detector_distance_config_max_profile_set(detector_config, ACC_CONFIG_PROFILE_3);
384  acc_detector_distance_config_reflector_shape_set(detector_config, ACC_DETECTOR_DISTANCE_REFLECTOR_SHAPE_GENERIC);
385  acc_detector_distance_config_peak_sorting_set(detector_config, ACC_DETECTOR_DISTANCE_PEAK_SORTING_STRONGEST);
386  acc_detector_distance_config_threshold_method_set(detector_config, ACC_DETECTOR_DISTANCE_THRESHOLD_METHOD_CFAR);
387  acc_detector_distance_config_threshold_sensitivity_set(detector_config, 0.5f);
388  acc_detector_distance_config_signal_quality_set(detector_config, 20.0f);
389  acc_detector_distance_config_close_range_leakage_cancellation_set(detector_config, false);
390  break;
391  }
392 }
